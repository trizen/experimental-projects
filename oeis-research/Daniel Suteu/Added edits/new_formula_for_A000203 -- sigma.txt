a(n) = Sum_{k=1..n} gcd(n, k) / phi(n / gcd(n, k)), where phi(k) is the Euler totient function. - ~~~~


func foo(n) {
    sum(1..n, {|k|
        gcd(n, k) / euler_phi(n / gcd(n, k))
    })
}

say 50.of(foo)
